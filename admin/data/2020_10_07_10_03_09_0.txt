package com.example.testslidepagefragmentproject;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.view.View;
import android.view.animation.Animation;
import android.view.animation.AnimationUtils;
import android.widget.Button;
import android.widget.FrameLayout;
import android.widget.LinearLayout;

public class MainActivity extends AppCompatActivity {
    private Button btnScreen1;
    private Button btnScreen2;
    private Button btnScreen3;
    private Button btnMenu;
    private FrameLayout frameContent;
    private LinearLayout linearContent;
    private LinearLayout linearSlidingPage;

    private Screen1Sub fragment1;
    private Screen2Sub fragment2;
    private Screen3Sub fragment3;

    private Animation translate_left;
    private Animation translate_right;

    private Boolean btnflag = false;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        btnScreen1 = (Button)findViewById(R.id.btnScreen1);
        btnScreen2 = (Button)findViewById(R.id.btnScreen2);
        btnScreen3 = (Button)findViewById(R.id.btnScreen3);
        btnMenu = (Button)findViewById(R.id.btnMenu);
        frameContent = (FrameLayout)findViewById(R.id.frameContent);
        linearContent = (LinearLayout)findViewById(R.id.linearContent);
        linearSlidingPage = (LinearLayout)findViewById(R.id.linearSlidingPage);

        translate_left = AnimationUtils.loadAnimation(getApplicationContext(), R.anim.translate_left);
        translate_right = AnimationUtils.loadAnimation(getApplicationContext(), R.anim.translate_right);

        fragment1 = new Screen1Sub();
        fragment2 = new Screen2Sub();
        fragment3 = new Screen3Sub();
        changeScreen(1);

        linearSlidingPage.setVisibility(View.INVISIBLE);

        btnScreen1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                changeScreen(1);
            }
        });

        btnScreen2.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                changeScreen(2);
            }
        });

        btnScreen3.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                changeScreen(3);
            }
        });

        btnMenu.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                linearSlidingPage.bringToFront();
                handleBtnMenuListener(view);
            }
        });
    }

    private void changeScreen(int no) {
        switch(no){
            case 1 :
                getSupportFragmentManager().beginTransaction().replace(R.id.linearContent, fragment1).commit();
                break;
            case 2 :
                getSupportFragmentManager().beginTransaction().replace(R.id.linearContent, fragment2).commit();
                break;
            case 3 :
                getSupportFragmentManager().beginTransaction().replace(R.id.linearContent, fragment3).commit();
                break;
        }
    }


    private void handleBtnMenuListener(View view) {
        if(btnflag == true){
            linearSlidingPage.startAnimation(translate_right);
            btnMenu.setText("OPEN");
            btnflag = false;
        }else{
            linearSlidingPage.startAnimation(translate_left);
            btnMenu.setText("CLOSE");
            btnflag = true;
        }

    }
}
